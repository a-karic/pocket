def find_or_create(role_name)
  role = find_by_name(role_name)
  return role if role.present?

  role = Role.new
  role.name = role_name
  save(role)
end

def user_role?(user:, role:)
  return false if user.nil? || user.new_record?

  user.roles.include? role
end
